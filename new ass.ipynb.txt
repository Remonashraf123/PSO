    "import numpy as np\n",
    "\n",
    "# 1. Define the Objective Function from the image\n",
    "def objective_function(pos):\n",
    "    x, y = pos[0], pos[1]\n",
    "    return x*2 + y*2 - 4*x - 6*y\n",
    "\n",
    "\n",
    "num_particles = 3        # N=3\n",
    "max_iter = 3             # max_iter=3\n",
    "w = 0.5                  # Inertia weight\n",
    "c1 = 1.5                 # Cognitive coefficient\n",
    "c2 = 2.0                 # Social coefficient\n",
    "\n",
    "bounds_min = -10\n",
    "bounds_max = 10\n",
    "\n",
    "particles_pos = np.random.uniform(bounds_min, bounds_max, (num_particles, 2))\n",
    "particles_velocity = np.random.uniform(-1, 1, (num_particles, 2))\n",
    "\n",
    "pbest_pos = particles_pos.copy()\n",
    "pbest_scores = np.array([objective_function(p) for p in particles_pos])\n",
    "best_idx = np.argmin(pbest_scores)\n",
    "gbest_pos = pbest_pos[best_idx].copy()\n",
    "gbest_score = pbest_scores[best_idx]\n",
    "\n",
    "print(f\"{'Iter':<5} | {'Best X':<10} | {'Best Y':<10} | {'Best Score':<10}\")\n",
    "print(\"-\" * 50)\n",

    "for t in range(max_iter):\n",
    "    for i in range(num_particles):\n",
    "        r1 = np.random.random()\n",
    "        r2 = np.random.random()\n",
    "        inertia = w * particles_velocity[i]\n",
    "        cognitive = c1 * r1 * (pbest_pos[i] - particles_pos[i])\n",
    "        social = c2 * r2 * (gbest_pos - particles_pos[i])\n",
    "        \n",
    "        particles_velocity[i] = inertia + cognitive + social\n",
    "        particles_pos[i] = particles_pos[i] + particles_velocity[i]\n",
    "        current_score = objective_function(particles_pos[i])\n",
    "        if current_score < pbest_scores[i]:\n",
    "            pbest_scores[i] = current_score\n",
    "            pbest_pos[i] = particles_pos[i].copy()\n",
    "            if current_score < gbest_score:\n",
    "                gbest_score = current_score\n",
    "                gbest_pos = particles_pos[i].copy()\n",
